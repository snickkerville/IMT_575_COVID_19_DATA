In [27]:
X = df.copy()

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

#determing the best number of clusters

inertia = []

for k in range(1, 12):
    km=KMeans(n_clusters=k, random_state=14, n_init=10)
    km.fit(X_scaled)
    inertia.append(km.inertia_)

plt.plot(range(1, 12), inertia)
plt.xlabel("Number of Clusters")
plt.ylabel('Inertia')
plt.title('Elbow Method', fontsize=12, fontweight='bold')
plt.grid(linewidth=0.2)
plt.show()
